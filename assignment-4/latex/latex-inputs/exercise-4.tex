    \begin{algorithm}[H]
        \DontPrintSemicolon
        \SetAlgoLined
        \KwResult{$QR$ Factorization}
        \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
        \Input{$A$}
        \Output{$Q$, $R$}
        \BlankLine
            \For{ j in range(0,N)}{
                $q = A(:,j)$\;
                
                \For{ i in range(0,j)}{
                    $q = \text{ Gram-Schmidt Process} $
                }
                $Q(:,j) = q$
            }
            \For{ i in range(0,N)}{
                $Q = \text{ normalize vectors }$
            }
            $R = Q^{T}A$
        \caption{$QR$ Factorization subroutine}
    \end{algorithm}
