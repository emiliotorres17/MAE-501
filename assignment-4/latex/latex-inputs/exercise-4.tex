    \begin{algorithm}[H]
        \DontPrintSemicolon
        \SetAlgoLined
        \KwResult{Calculating $d$}
        \SetKwInOut{Input}{Input}\SetKwInOut{Output}{Output}
        \Input{$a$, $b$, $c$, $d$}
        \Output{$d$}
        \BlankLine
            \For{ i in range(1,N)}{
                $m = a[i]/b[i]$\;
                $b[i] = b[i]-m*c[i]$\;
                $d[i] = d[i]-m*d[i]$\;
            }
            $d[-1] = d[-1]/b[-1]$\;
            \For{ i in range(N-2,-1, -1)}{
                $d[i] = (d[i]-c[i]*d[i+1])/b[i]$\;
            }
        \caption{Tri-diagonal solver subroutine}
    \end{algorithm}
